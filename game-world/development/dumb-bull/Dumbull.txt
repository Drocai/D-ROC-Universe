    <script>
        // --- 1. PROCEDURAL AUDIO (Synthesizer) ---
        const AudioEngine = {
            ctx: null,
            init: function() {
                window.AudioContext = window.AudioContext || window.webkitAudioContext;
                this.ctx = new AudioContext();
            },
            createOsc: function(type, freq, start, end, vol=0.1) {
                if(!this.ctx) return;
                const osc = this.ctx.createOscillator();
                const gain = this.ctx.createGain();
                osc.type = type;
                osc.frequency.setValueAtTime(freq, start);
                gain.gain.setValueAtTime(vol, start);
                gain.gain.exponentialRampToValueAtTime(0.001, end);
                osc.connect(gain).connect(this.ctx.destination);
                osc.start(start);
                osc.stop(end);
            },
            smash: function() {
                if(!this.ctx) return;
                const t = this.ctx.currentTime;
                // Noise burst
                const bufferSize = this.ctx.sampleRate * 0.3;
                const buffer = this.ctx.createBuffer(1, bufferSize, this.ctx.sampleRate);
                const data = buffer.getChannelData(0);
                for (let i = 0; i < bufferSize; i++) data[i] = Math.random() * 2 - 1;
                const noise = this.ctx.createBufferSource();
                noise.buffer = buffer;
                const filter = this.ctx.createBiquadFilter();
                filter.type = 'highpass';
                filter.frequency.setValueAtTime(1000, t);
                filter.frequency.exponentialRampToValueAtTime(100, t + 0.2);
                const gain = this.ctx.createGain();
                gain.gain.setValueAtTime(0.5, t);
                gain.gain.exponentialRampToValueAtTime(0.01, t + 0.3);
                noise.connect(filter).connect(gain).connect(this.ctx.destination);
                noise.start();
                // Underlying bass kick
                this.createOsc('square', 80, t, t+0.2, 0.3);
            },
            collect: function() {
                if(!this.ctx) return;
                const t = this.ctx.currentTime;
                this.createOsc('sine', 1200, t, t+0.1, 0.1);
                this.createOsc('sine', 1800, t+0.05, t+0.2, 0.1);
            },
            dash: function() {
                if(!this.ctx) return;
                const t = this.ctx.currentTime;
                const osc = this.ctx.createOscillator();
                const gain = this.ctx.createGain();
                osc.type = 'sawtooth';
                osc.frequency.setValueAtTime(100, t);
                osc.frequency.linearRampToValueAtTime(600, t + 0.2);
                gain.gain.setValueAtTime(0.1, t);
                gain.gain.linearRampToValueAtTime(0, t + 0.2);
                const filter = this.ctx.createBiquadFilter();
                filter.type = 'lowpass';
                filter.frequency.value = 500;
                osc.connect(filter).connect(gain).connect(this.ctx.destination);
                osc.start();
                osc.stop(t+0.2);
            }
        };

        // --- 2. SETUP & OPTIMIZATION ---
        const canvas = document.getElementById('gameCanvas');
        const ctx = canvas.getContext('2d', { alpha: false }); // Optimize: No alpha needed on backbuffer
        
        let width, height;
        let bgCanvas; // Offscreen canvas for static background

        function resize() {
            width = window.innerWidth;
            height = window.innerHeight;
            canvas.width = width;
            canvas.height = height;
            preRenderBackground();
        }
        
        // --- 3. OFFSCREEN RENDERING (Performance) ---
        function preRenderBackground() {
            bgCanvas = document.createElement('canvas');
            bgCanvas.width = width;
            bgCanvas.height = height;
            const bCtx = bgCanvas.getContext('2d');

            // Dark Floor
            bCtx.fillStyle = '#0f1219';
            bCtx.fillRect(0, 0, width, height);

            // Grid Pattern
            bCtx.strokeStyle = '#1a2639';
            bCtx.lineWidth = 1;
            const gridSize = 80;
            
            bCtx.beginPath();
            // Vertical lines
            for (let x = 0; x <= width; x += gridSize) {
                bCtx.moveTo(x, 0); bCtx.lineTo(x, height);
            }
            // Horizontal lines
            for (let y = 0; y <= height; y += gridSize) {
                bCtx.moveTo(0, y); bCtx.lineTo(width, y);
            }
            bCtx.stroke();

            // Vignette
            const grad = bCtx.createRadialGradient(width/2, height/2, height/2, width/2, height/2, height);
            grad.addColorStop(0, 'rgba(0,0,0,0)');
            grad.addColorStop(1, 'rgba(0,0,0,0.6)');
            bCtx.fillStyle = grad;
            bCtx.fillRect(0,0,width,height);
        }

        window.addEventListener('resize', resize);
        
        // --- 4. GAME ENTITIES ---

        class Bull {
            constructor() {
                this.x = window.innerWidth/2;
                this.y = window.innerHeight/2;
                this.vx = 0; this.vy = 0;
                this.size = 40; // Collision Radius
                this.angle = 0;
            }
            update(jx, jy) {
                // Acceleration
                this.vx += jx * 1.5;
                this.vy += jy * 1.5;
                // Friction
                this.vx *= 0.92;
                this.vy *= 0.92;
                // Pos
                this.x += this.vx;
                this.y += this.vy;
                
                // Screen Wrap
                if(this.x < -this.size) this.x = width + this.size;
                if(this.x > width + this.size) this.x = -this.size;
                if(this.y < -this.size) this.y = height + this.size;
                if(this.y > height + this.size) this.y = -this.size;

                // Rotation based on movement
                if(Math.abs(this.vx) > 0.1 || Math.abs(this.vy) > 0.1) {
                    const targetAngle = Math.atan2(this.vy, this.vx);
                    // Smooth rotation lerp could go here, but instant is snappy for arcade
                    this.angle = targetAngle;
                }
            }
            dash() {
                const speed = Math.hypot(this.vx, this.vy) || 1;
                this.vx = (this.vx/speed) * 35;
                this.vy = (this.vy/speed) * 35;
                shake = 10; // Screen shake on dash
                AudioEngine.dash();
                
                // Dash Trail
                for(let i=0; i<5; i++) {
                    particles.push(new Particle(this.x, this.y, '#ff0055', -this.vx*0.5, -this.vy*0.5));
                }
            }
            draw() {
                ctx.save();
                ctx.translate(this.x, this.y);
                ctx.rotate(this.angle);

                // Body (Vector)
                ctx.fillStyle = '#00f0ff';
                ctx.shadowColor = '#00f0ff';
                ctx.shadowBlur = 15;
                
                ctx.beginPath();
                ctx.moveTo(20, 0); // Nose
                ctx.lineTo(-15, 15); // Back Left
                ctx.lineTo(-10, 0); // Inner Notch
                ctx.lineTo(-15, -15); // Back Right
                ctx.closePath();
                ctx.fill();

                // Horns
                ctx.fillStyle = '#ffffff';
                ctx.beginPath();
                ctx.moveTo(10, 5); ctx.lineTo(25, 15); ctx.lineTo(10, 10);
                ctx.moveTo(10, -5); ctx.lineTo(25, -15); ctx.lineTo(10, -10);
                ctx.fill();

                ctx.restore();
            }
        }

        class Target {
            constructor(type) {
                this.type = type; // 0 = China, 1 = Apple
                this.respawn();
            }
            respawn() {
                this.active = true;
                this.r = this.type === 0 ? 25 : 15;
                this.x = Math.random() * (width - 100) + 50;
                this.y = Math.random() * (height - 100) + 50;
                // Random polygon shape for china
                this.polySides = Math.floor(Math.random() * 3) + 4; // 4 to 6 sides
                this.rot = Math.random() * Math.PI;
            }
            draw() {
                if(!this.active) return;
                ctx.save();
                ctx.translate(this.x, this.y);
                
                if (this.type === 0) {
                    // China (Glass/Ceramic)
                    ctx.strokeStyle = '#ecf0f1';
                    ctx.lineWidth = 3;
                    ctx.shadowColor = '#ecf0f1';
                    ctx.shadowBlur = 5;
                    ctx.rotate(this.rot);
                    
                    ctx.beginPath();
                    for(let i=0; i<this.polySides; i++) {
                        const theta = (i / this.polySides) * Math.PI * 2;
                        const px = Math.cos(theta) * this.r;
                        const py = Math.sin(theta) * this.r;
                        if(i===0) ctx.moveTo(px, py);
                        else ctx.lineTo(px, py);
                    }
                    ctx.closePath();
                    ctx.stroke();
                    // Inner fill opacity
                    ctx.fillStyle = 'rgba(236, 240, 241, 0.1)';
                    ctx.fill();
                } else {
                    // Apple (Energy Orb)
                    ctx.fillStyle = '#f1c40f';
                    ctx.shadowColor = '#f1c40f';
                    ctx.shadowBlur = 20;
                    ctx.beginPath();
                    ctx.arc(0, 0, this.r, 0, Math.PI*2);
                    ctx.fill();
                    
                    // Pulse
                    ctx.beginPath();
                    ctx.arc(0, 0, this.r + Math.sin(Date.now()/100)*3, 0, Math.PI*2);
                    ctx.strokeStyle = 'rgba(241, 196, 15, 0.5)';
                    ctx.stroke();
                }
                ctx.restore();
            }
        }

        class Particle {
            constructor(x, y, color, vx, vy) {
                this.x = x; this.y = y;
                this.vx = vx || (Math.random() - 0.5) * 15;
                this.vy = vy || (Math.random() - 0.5) * 15;
                this.life = 1.0;
                this.color = color;
                this.size = Math.random() * 4 + 2;
            }
            update() {
                this.x += this.vx;
                this.y += this.vy;
                this.vx *= 0.95; // Drag
                this.vy *= 0.95;
                this.life -= 0.03;
            }
            draw() {
                ctx.globalAlpha = this.life;
                ctx.fillStyle = this.color;
                // Additive blending for glowy look
                ctx.globalCompositeOperation = 'lighter'; 
                ctx.beginPath();
                ctx.arc(this.x, this.y, this.size, 0, Math.PI*2);
                ctx.fill();
                ctx.globalCompositeOperation = 'source-over'; // Reset
                ctx.globalAlpha = 1.0;
            }
        }

        // --- 5. GAME LOOP & STATE ---
        const player = new Bull();
        const targets = Array.from({length: 8}, () => new Target(0));
        let particles = [];
        let score = 0;
        let streak = 0;
        let running = false;
        let shake = 0; // Screen shake magnitude

        // Joystick Input
        let joyX = 0, joyY = 0;
        const joyOuter = document.getElementById('joystickOuter');
        const joyInner = document.getElementById('joystickInner');
        
        // Touch Handlers
        const handleJoy = (e) => {
            e.preventDefault();
            const touch = e.touches ? e.touches[0] : e;
            const rect = joyOuter.getBoundingClientRect();
            const centerX = rect.left + rect.width/2;
            const centerY = rect.top + rect.height/2;
            let dx = touch.clientX - centerX;
            let dy = touch.clientY - centerY;
            const dist = Math.min(Math.hypot(dx, dy), rect.width/2);
            const angle = Math.atan2(dy, dx);
            joyX = Math.cos(angle) * (dist/(rect.width/2));
            joyY = Math.sin(angle) * (dist/(rect.width/2));
            
            // UI Update
            const uiDist = Math.min(dist, (rect.width/2)-30);
            joyInner.style.transform = `translate(calc(-50% + ${Math.cos(angle)*uiDist}px), calc(-50% + ${Math.sin(angle)*uiDist}px))`;
        };
        const resetJoy = () => { joyX=0; joyY=0; joyInner.style.transform = `translate(-50%, -50%)`; };

        joyOuter.addEventListener('touchstart', handleJoy, {passive:false});
        joyOuter.addEventListener('touchmove', handleJoy, {passive:false});
        joyOuter.addEventListener('touchend', resetJoy);
        
        // Dash
        const dashBtn = document.getElementById('dashBtn');
        let canDash = true;
        dashBtn.addEventListener('touchstart', (e) => {
            e.preventDefault();
            if(running && canDash) {
                player.dash();
                canDash = false;
                dashBtn.classList.add('cooldown');
                setTimeout(() => { canDash=true; dashBtn.classList.remove('cooldown'); }, 800);
            }
        });

        // Main Loop
        function loop() {
            if(!running) return;

            // 1. CLEAR & BACKGROUND
            ctx.setTransform(1,0,0,1,0,0); // Reset transform
            // Apply Shake
            if(shake > 0) {
                const rx = (Math.random()-0.5) * shake;
                const ry = (Math.random()-0.5) * shake;
                ctx.translate(rx, ry);
                shake *= 0.9;
                if(shake < 0.5) shake = 0;
            }
            
            // Fast Background Blit
            ctx.drawImage(bgCanvas, 0, 0);

            // 2. PLAYER LOGIC
            player.update(joyX, joyY);
            player.draw();

            // 3. ENTITY LOGIC
            targets.forEach(t => {
                t.draw();
                if(t.active) {
                    const dx = player.x - t.x;
                    const dy = player.y - t.y;
                    const dist = Math.hypot(dx, dy);
                    
                    if(dist < player.size + t.r) {
                        t.active = false;
                        
                        if(t.type === 0) { // SMASH
                            AudioEngine.smash();
                            shake = 15;
                            streak++;
                            score += 50 * streak;
                            document.getElementById('streakVal').innerText = "x" + streak;
                            showCombo();
                            
                            // Debris
                            for(let i=0; i<8; i++) particles.push(new Particle(t.x, t.y, '#ecf0f1'));
                            // Spawn Apple Chance
                            if(Math.random() > 0.8) targets.push(new Target(1));
                            
                            setTimeout(() => t.respawn(), 1200);
                        } else { // COLLECT
                            AudioEngine.collect();
                            score += 500;
                            // Gold Particles
                            for(let i=0; i<12; i++) particles.push(new Particle(t.x, t.y, '#f1c40f'));
                        }
                        
                        document.getElementById('scoreVal').innerText = score;
                    }
                }
            });

            // 4. PARTICLES
            for(let i=particles.length-1; i>=0; i--) {
                let p = particles[i];
                p.update();
                p.draw();
                if(p.life <= 0) particles.splice(i, 1);
            }

            // Streak Decay
            if(streak > 0 && Math.random() < 0.005) {
                streak = 0;
                document.getElementById('streakVal').innerText = "0";
            }

            requestAnimationFrame(loop);
        }

        const comboEl = document.getElementById('comboTxt');
        function showCombo() {
            comboEl.innerText = ["SMASH!", "CRUSH!", "WRECK!", "BOOM!"][Math.floor(Math.random()*4)] + " x" + streak;
            comboEl.style.opacity = 1;
            comboEl.style.transform = "translate(-50%, -50%) scale(1.5)";
            setTimeout(() => {
                comboEl.style.opacity = 0;
                comboEl.style.transform = "translate(-50%, -50%) scale(1.0)";
            }, 500);
        }

        // Init
        resize();
        document.getElementById('startBtn').addEventListener('click', () => {
            AudioEngine.init();
            document.getElementById('overlay').style.display = 'none';
            running = true;
            loop();
        });

    </script>
